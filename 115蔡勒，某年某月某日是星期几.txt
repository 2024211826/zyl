#include <iostream>
int zeller(int day, int month, int year) {
    if (month < 3) {
        month += 12;
        year -= 1;
    }
    int c = year / 100;
    int y = year % 100;
    int w = (c / 4 - 2 * c + y + y / 4 + 13 * (month + 1) / 5 + day - 1) % 7;
    return w;
}
std::string dayOfWeekName(int dayOfWeek) {
    switch (dayOfWeek) {
    case 0: return "Sunday";
    case 1: return "Monday";
    case 2: return "Tuesday";
    case 3: return "Wednesday";
    case 4: return "Thursday";
    case 5: return "Friday";
    case 6: return "Saturday";
    default: return "Unknown";
    }
}
int main() {
    int day, month, year;
    std::cout << "Enter date (day month year): ";
    std::cin >> day >> month >> year;
    int dayOfWeek = zeller(day, month, year);
    std::cout << "The day of the week is: " << dayOfWeekName(dayOfWeek) << std::endl;
    return 0;
}
